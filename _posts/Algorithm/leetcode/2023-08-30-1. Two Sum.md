```python
class Solution:
    def twoSum(self,nums: List[int], target: int) -> List[int]:
        l = len(nums)
        a = [0,0]
        if(l>=3):
            for i in range(l-1):
                for j in range(i+1,l):
                    x = nums[i] + nums[j]
                    if(target==x):
                        a[0] = i
                        a[1] = j
        else:
            a[0] = 0
            a[1] = 1    

        return a    
```
