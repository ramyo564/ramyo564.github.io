class Solution:
    def minSubArrayLen(self, target: int, nums: List[int]) -> int:
        nums.sort()
        if target in nums:
            return 1
        if sum(nums) == target:
            return len(nums)
        if len(nums) < 2:
            return 0
        num_dict = {}

        for i, num in enumerate(nums):
            complement = target - num
            if num >= complement or complement in num_dict:
                return 2
            num_dict[i] = num

        last_key = list(num_dict.keys())[-1]

        new_target = num_dict[last_key-1] + num_dict[last_key]
        count = 3
        for i in range(last_key-2, 0, -1):
            if new_target + num_dict[i] >= target:
                return count
            else: 
                count +=1
                new_target += num_dict[i]

        return 0
```

class Solution:
    def minSubArrayLen(self, target: int, nums: List[int]) -> int:
        if sum(nums) < target: return 0

        s, l, count = 0, 0, len(nums)
        for r, val in enumerate(nums):
            s += val
            while s >= target:
                s -= nums[l]
                count = min(count, r - l + 1)
                l += 1
                
        return count
```
