## Node.js

function solution(n, k) {
    var z = parseInt(n/10) * 2000
    return n * 12000 + k * 2000 - z;
}

function solution(n, k) {
    return n*12000 + k*2000 - parseInt(n/10)*2000

// 다른 사람 풀이 2
